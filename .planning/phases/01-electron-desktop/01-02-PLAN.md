```javascript
export function getWeeklyRequired(goal) {
  if (!goal.targetDate) return null
  const now = new Date()
  const target = new Date(goal.targetDate)
  if (target <= now) return null  // target date already passed
  const msPerWeek = 7 * 24 * 60 * 60 * 1000
  const weeksRemaining = Math.max(1, (target - now) / msPerWeek)
  const remaining = goal.target - (goal.saved || 0)
  if (remaining <= 0) return 0   // already fully funded
  return remaining / weeksRemaining  // float; caller rounds for display
}
```

```javascript
function FolderWatcherRow() {
  const [watchedFolder, setWatchedFolder] = useState(() => {
    return localStorage.getItem('budgetpulse_watchfolder') || null
  })

  async function handlePickFolder() {
    const folder = await window.electronAPI?.pickFolder()
    if (!folder) return
    setWatchedFolder(folder)
    localStorage.setItem('budgetpulse_watchfolder', folder)
    await window.electronAPI?.watchFolder(folder)
  }

  if (!window.electronAPI) return null  // hide entirely in web-mode dev

  return (
    <div className="mx-5 mb-4 glass rounded-xl px-4 py-3 flex items-center justify-between gap-3">

        <p className="text-xs font-medium text-[var(--color-text-bright)]">Watch folder</p>
        <p className="text-xs text-[var(--color-text-muted)] truncate">
          {watchedFolder || 'No folder selected'}
        </p>
      </div>
      <button
        onClick={handlePickFolder}
        className="shrink-0 text-xs px-3 py-1.5 rounded-lg"
        style={{ background: 'var(--color-surface-3)', color: 'var(--color-accent)', border: '1px solid var(--color-border)' }}
      >
        {watchedFolder ? 'Change' : 'Select'}
      </button>
    </div>
  )
}
```

```javascript
  async function handleBuffer(buffer, filePath) {
    setPhase('parsing')
    setError(null)

      // parsePDF currently accepts a File object. We need to parse from ArrayBuffer directly.
      // Create a minimal File-like object or use parsePDF's internal flow.

      const fakeFile = new File([buffer], filePath.split('/').pop() || 'statement.pdf', { type: 'application/pdf' })

        setPhase('drop')

      setPhase('review')
    } catch (e) {

    }
  }
```
